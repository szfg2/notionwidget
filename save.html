<body>

    <label for="dateOfOnset">Date of Onset:</label>

    <input type="date" id="dateOfOnset" name="dateOfOnset">

    <button onclick="calculateDays()">Submit</button>

    <p id="output"></p>

    <script>

        // JavaScript function placed here for direct invocation

        function calculateDays() {

            const dateOfOnset = document.getElementById('dateOfOnset').value;

            const date1 = new Date(dateOfOnset);

            const date2 = new Date(); // Current date

            const diffTime = Math.abs(date2 - date1);

            const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24)); 

            document.getElementById('output').innerHTML = "Day " + diffDays + " of stroke<br>- Date of onset: " + dateOfOnset;

        }

    </script>

</body>
